## Key files

**`novu/workflows.js`** — defines the `slack-notification` workflow with a `send-to-slack` chat step returning `"Hello from Novu!"`:
```js
const { workflow } = require('@novu/framework');

const slackNotificationWorkflow = workflow('slack-notification', async ({ step }) => {
  await step.chat('send-to-slack', async () => {
    return { body: 'Hello from Novu!' };
  });
});

module.exports = { slackNotificationWorkflow };
```

**`index.js`** — mounts the Novu Bridge handler at `/api/novu` and listens on port 3000:
```js
const express = require('express');
const { serve } = require('@novu/framework/express');
const { slackNotificationWorkflow } = require('./novu/workflows');

const app = express();
app.use(express.json());
app.use('/api/novu', serve({ workflows: [slackNotificationWorkflow] }));
app.listen(3000, () => { ... });
```
